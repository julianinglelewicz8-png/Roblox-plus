function playLoudZombie() {
    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const gain = audioCtx.createGain();
    gain.gain.value = 1.0;
    gain.connect(audioCtx.destination);

    const oscillator = audioCtx.createOscillator();
    oscillator.type = 'square';
    oscillator.frequency.setValueAtTime(120, audioCtx.currentTime);

    const noiseBuffer = audioCtx.createBuffer(1, audioCtx.sampleRate * 0.3, audioCtx.sampleRate);
    const data = noiseBuffer.getChannelData(0);
    for (let i = 0; i < data.length; i++) {
        data[i] = (Math.random() * 2 - 1) * 0.6;
    }
    const noise = audioCtx.createBufferSource();
    noise.buffer = noiseBuffer;

    const noiseGain = audioCtx.createGain();
    noiseGain.gain.value = 0.7;
    noise.connect(noiseGain).connect(gain);

    oscillator.connect(gain);
    oscillator.start();
    noise.start();

    oscillator.stop(audioCtx.currentTime + 0.45);
    noise.stop(audioCtx.currentTime + 0.45);
}

document.getElementById('startBtn').addEventListener('click', () => {
    document.body.classList.add('active');
    document.getElementById('startBtn').disabled = true;
    playLoudZombie();
    setTimeout(() => {
        document.getElementById('hack').style.display = 'none';
        document.getElementById('joke').style.display = 'block';
    }, 4000);
});
